51 {
52  auto offset = buffer.size();
53  buffer.push_back(static_cast<uint8_t>(opcode));
54  buffer.resize(offset + sizeof(GLOpcode) + sizeof(T));
55  return reinterpret_cast<T*>(&buffer[offset + sizeof(GLOpcode)]);
